# Create a pie chart
plt.pie(
    # using this column for plotting
    df_ptct['dentists'],
    
    # with the labels being the column data_description
    labels=df_ptct['data_description'],
    
    # with no shadows
    shadow=False,
    
    # with colors
    colors=colors,
    
    # with one slide exploded out
    #explode=(0, 0, 0.12, 0),
    
    # You can change the value of the angle based on your data. 
    #The goal is to reduce the space the text labels will take.
    startangle=180,
    
    # with the percent listed as a fraction
    autopct='%1.1f%%',
    )

plt.title('Insurance Types')
# plt.tight_layout()
#plt.title('Medicaid and CHIP Participation Rate')
plt.axis('equal')
plt.savefig('CO_Insurance_Participation.png', dpi=100)
